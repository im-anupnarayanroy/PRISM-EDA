# CONCEPT — The DataFrame

A DataFrame is a 2D labeled table. Think of it as a super-powered Excel sheet living in Python memory. Rows are indexed (0, 1, 2 …), columns have names, and every column holds values of the same data type.

Filtering & Selecting Data

Slice rows and columns using boolean masks, .loc[], .iloc[], and query strings.

# CONCEPT — Boolean Masking

df[condition] returns only rows where the condition is True. Chain conditions with & (AND), | (OR), ~ (NOT). Always wrap each condition in parentheses when combining.
boolean_filter.py
# Single condition skincare = df[df['Product type'] == 'skincare'] # Multi-condition: high-revenue skincare mask = ( (df['Product type'] == 'skincare') & (df['Revenue generated'] > 8000) & (df['Defect rates'] < 1.0) ) result = df[mask] print(result.shape)
⚡Output

Grouping & Aggregation

Use .groupby() to segment data and calculate group-level statistics — the core of supply chain analytics.

# CONCEPT — Split → Apply → Combine

groupby() works in 3 steps: Split the DataFrame into groups → Apply a function to each group → Combine results into a new DataFrame. This pattern handles 80% of real analytics queries.
groupby_basic.py
# Revenue by Product type df.groupby('Product type')['Revenue generated'].mean().round(2) # Multiple aggregations at once df.groupby('Product type').agg({ 'Revenue generated': ['mean', 'sum'], 'Defect rates' : 'mean', 'Stock levels' : 'sum' }).round(2)

Rename columns, handle missing values, add derived features, and reshape DataFrames into the right form for your API.

missing_values.py
# Check missing values df.isnull().sum()[df.isnull().sum() > 0] # Fill with group mean (smarter) df['mfg_cost'] = df.groupby('product_type')['mfg_cost'].transform( lambda x: x.fillna(x.mean()) ) # Drop rows where critical cols are null df.dropna(subset=['SKU', 'product_type']) # Fill categorical with mode df['Location'].fillna(df['Location'].mode()[0])
⚡Output

from fastapi import FastAPI, Query, HTTPException from pydantic import BaseModel import pandas as pd from typing import Optional, List app = FastAPI(title="Supply Chain Analytics") # ── Load once at startup ────────────────────────────── df = pd.read_csv("supply_chain_data.csv") df.columns = ( df.columns.str.lower().str.replace(' ', '_') ) # snake_case all column names # ── Schemas ─────────────────────────────────────────── class CustomQuery(BaseModel): group_by : str metric : str # e.g. "revenue_generated" agg_func : str = "mean" # sum | mean | count | max top_n : Optional[int] = None # ── Routes ──────────────────────────────────────────── @app.get("/health") def health(): return {"status": "ok", "rows": len(df), "cols": len(df.columns)} @app.get("/products") def list_products( product_type : Optional[str] = Query(None), location : Optional[str] = Query(None), min_revenue : Optional[float] = Query(None), limit : int = Query(20, le=100) ): fdf = df.copy() if product_type: fdf = fdf[fdf.product_type == product_type] if location: fdf = fdf[fdf.location == location] if min_revenue: fdf = fdf[fdf.revenue_generated >= min_revenue] return fdf.head(limit).to_dict(orient="records") @app.get("/products/{sku}") def get_product(sku: str): row = df[df.str.upper() == sku.upper()] if row.empty(): raise HTTPException(404, f"SKU {sku} not found") return row.iloc[0].to_dict() @app.get("/analytics/summary") def summary(): return { "total_skus" : int(len(df)), "total_revenue" : round(float(df.revenue_generated.sum()), 2), "avg_defect_rate": round(float(df.defect_rates.mean()), 3), "pass_rate" : round((df.inspection_results=="Pass").mean()*100, 1), "locations" : df.location.unique().tolist(), "product_types" : df.product_type.unique().tolist(), } @app.get("/analytics/by-location") def by_location(): result = df.groupby("location").agg( total_revenue = ("revenue_generated", "sum"), avg_defect = ("defect_rates", "mean"), sku_count = ("sku", "count"), ).reset_index().round(2) return result.to_dict(orient="records") @app.get("/analytics/top-products") def top_products( n : int = Query(10, le=50), metric : str = Query("revenue_generated"), asc : bool = Query(False) ): if metric not in df.select_dtypes("number").columns: raise HTTPException(400, f"{metric} is not a numeric column") fn = df.nsmallest if asc else df.nlargest return fn(n, metric).to_dict(orient="records") @app.get("/analytics/defect-risk") def defect_risk(threshold: float = Query(3.5)): risky = df[df.defect_rates > threshold][ ["sku","product_type","supplier_name","defect_rates","inspection_results"] ].sort_values("defect_rates", ascending=False) return {"threshold": threshold, "at_risk": risky.to_dict(orient="records")} @app.post("/analytics/custom-query") def custom_query(q: CustomQuery): try: result = ( df.groupby(q.group_by)[q.metric] .agg(q.agg_func) .reset_index() .sort_values(q.metric, ascending=False) ) if q.top_n: result = result.head(q.top_n) return result.to_dict(orient="records") except Exception as e: raise HTTPException(400, str(e))
